\section{Introduction}
The longest increasing subsequence (LIS for short) problem is a classic algorithmic problem with applications in areas such as sequence analysis, pattern discovery, and optimization. Given a sequence of numbers, the task is to find a longest subsequence whose elements appear in strictly increasing order while preserving their original order, or to compute length of such sequence.
Although the LIS problem has been studied extensively in many variants, the trade-off between time and space remains relevant when the input is large or memory is limited. In this paper, we focus on computing both the length of an LIS and an actual LIS using small auxiliary space. First contribution regards a tighter analysis of the Patience Sorting based approach introduced by Kiyomi et al.~\cite{kiyomi2017space}, showing that the length can be computed in $O\!\left(\max(n \log n, n^2/s)\right)$ time. We also provide an alternative divide-and-conquer method for recovering the sequence, which combined with the improved length analysis shows an algorithm working in $O\!\left(\max(n \log n \log \log s, n^2/s)\right)$ time.


\subsection{Preliminaries}
\subsubsection{LIS problem}
We consider the standard longest increasing subsequence problem on an input sequence $A=(a_1,\ldots,a_n)$ of comparable elements.

\begin{definition}
We define basic notions as follows:
\begin{itemize}
    \item An \emph{increasing subsequence} is a sequence of indices $1 \le i_1 < \cdots < i_\ell \le n$ such that
    \[
    a_{i_1} < a_{i_2} < \cdots < a_{i_\ell}.
    \]
    \item A \emph{longest increasing subsequence} of $A$ is denoted by $L(A)$; if multiple optimal subsequences exist, $L(A)$ can represent any one of them.
    \item The length of $L(A)$:
    \[
    |L(A)|=\max\left\{\ell : \exists\, i_1 < \cdots < i_\ell | a_{i_1} < \cdots < a_{i_\ell}\right\}.
    \]
\end{itemize}
There are two usual problems; we can either ask for computing $L(A)$, or just its length $|L(a)|$
\end{definition}

We use the RAM model, where the input sequence is read-only and the algorithm has $O(s)$ auxiliary words.

\section{New results}
\subsection{Length computation}
We provide a tighter analysis of the Patience Sorting algorithm presented in \cite{kiyomi2017space}. We show that the algorithm runs in $O\!\left(\max(n \log n, \frac{n^2}{s})\right)$ time, improving the previous $O(\frac{n^2}{s} \log n)$ bound.

\begin{definition}
Let us define the parameters of Patience Sort as follows:
\begin{itemize}
    \item $K$ -- the number of passes over the input.
    \item $j_i$ (for $i \in \{1,\ldots,K\}$) -- the index of the largest pile kept in memory after the $i$-th pass, with $j_0=0$.
    \item $W_i$ -- the number of expensive filter operations during the $i$-th pass.
\end{itemize}
Then the running time can be written as:
\[
T(n,s) = O\!\left(nK + \log(s)\sum_{i=1}^K W_i\right).
\]
\end{definition}

\begin{claim} \label{claim:linear-filters}
The total number of expensive filter operations is linear, i.e.,
\[
\sum_{i=1}^K W_i = O(n).
\]
\end{claim}

\begin{proof}
For each pass $i \in \{1, \dots, K\}$, let us define the active interval $I_i = (j_{i-1}, j_{i-1}+2s]$.

Fix a pass $i$. During the first phase of this pass, the expensive filter is invoked only for elements whose pile index belongs to $I_i$. Once the new boundary $j_i$ is determined, the second phase invokes the expensive filter only if the pile index is in $(j_{i-1}, j_i]$. By the properties of the algorithm, we know that $j_i \le j_{i-1} + 2s$, which means $(j_{i-1}, j_i] \subseteq I_i$. Thus, we can see that:
\[
W_i \le \sum_{k \in I_i} |P_k| + \sum_{k=j_{i-1}+1}^{j_i} |P_k| \le 2\sum_{k \in I_i} |P_k|.
\]
This gives us: 
\[
\sum_{i=1}^K W_i \le 2\sum_{i=1}^K \sum_{k \in I_i} |P_k|.
\]

By \cite{kiyomi2017space} we have $j_i \ge j_{i-1}+s$. This implies that the left endpoints of the intervals $I_i$ advance by at least $s$ in each pass. Since each interval $I_i$ has a width of exactly $2s$, any pile index $k$ can belong to at most two intervals $I$. Therefore, summing the pile sizes over all intervals yields at most twice the total number of elements:
\[
\sum_{i=1}^K \sum_{k \in I_i} |P_k| \le 2\sum_{k=1}^n |P_k| = 2n.
\]
Thus, we get $\sum_{i=1}^K W_i \le 4n = O(n)$.
\end{proof}

It is established in \cite{kiyomi2017space} that $K = O(\frac{n}{s})$, which gives us the total running time:
\[
T(n,s) = O\!\left(\frac{n^2}{s} + \log{s}*O(n)\right) = O\!\left(max(n\log{n}, \frac{n^2}{s})\!\right).
\]
\subsubsection{LIS computation}
To recover an LIS, we present an alternative divide-and-conquer algorithm which, combined with the improved length bound, runs in $O\!\left(\max(n \log n \log \log s, n^2/s)\right)$ time. This improves on the earlier $O(n^2 \log^2(s) / s)$ result.
